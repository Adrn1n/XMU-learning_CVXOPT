        \alpha & 4      \\
    \end{vmatrix}      & = 4 - \alpha^2 > 0, \ \alpha \in (-2, 2)                            \\
    \begin{vmatrix}
        1      & \alpha & -1 \\
        \alpha & 4      & 2  \\
        -1     & 2      & 4  \\
    \end{vmatrix} & = 1 \times 4 \times 4 + (-1) \times 2 \alpha + (-1) \times 2 \alpha      \\

                            & = -4 (\alpha^2 + \alpha -2) > 0, \ \alpha \in (-2, 1)          \\
    \\
    \therefore \alpha       & \in (-2, 1)                                                    \\
\end{align*}
