    \because \operatorname{dom}{f}, \operatorname{dom}{g} & \text{ are convex}                                 \\
    \therefore \operatorname{dom}{h}                      & \text{ is convex}                                  \\

    h(k x + (1 - k) y)                                    & = f(k x + (1 - k) y) + g(k x + (1 - k) y)          \\

\end{align*}

    f(k (x_1, y_1) + (1 - k) (x_2,y_2)) & = f(k x_1 + (1 - k) x_2,k y_1 + (1 - k) y_2)                    \\
                                        & = (k x_1 + (1 - k) x_2) \log{(k x_1 + (1 - k) x_2)}             \\
                                        & \phantom{=} + (k y_1 + (1 - k) y_2) \log{(k y_1 + (1 - k) y_2)} \\
                                        & \le (k x_1 \log{x_1} + (1 - k) x_2 \log{x_2})                   \\
                                        & \phantom{=} + (k y_1 \log{y_1} + (1 - k) y_2 \log{y_2})         \\
                                        & = k (x_1 \log{x_1} + y_1 \log{y_1})                             \\

                                        & = k f(x_1, y_1) + (1 - k) f(x_2, y_2)                           \\
    \text{Q.}                           & \text{E.D.}                                                     \\
\end{align*}
